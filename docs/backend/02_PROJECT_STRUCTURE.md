## 2. Naming

- File: `kebab-case.layer.ts` (e.g., `auth.controller.ts`, `plan.service.ts`).
- Class: `PascalCase`.
- Function & variable: `camelCase`.
- Const env: `SCREAMING_SNAKE_CASE`.
- Tabel DB: `snake_case`, plural (`users`, `weekly_plans`).

## 3. Pattern: Controller → Service → Repository

**Controller** thin, hanya parsing request dan kirim response:
```ts
// controllers/plan.controller.ts
import { Request, Response, NextFunction } from 'express';
import { planService } from '../services/plan.service';
import { asyncHandler } from '../middleware/async.middleware';

export const planController = {
  generate: asyncHandler(async (req: Request, res: Response) => {
    const userId = req.user!.id;
    const plan = await planService.generateForUser(userId);
    res.status(201).json({ plan });
  }),

  getCurrent: asyncHandler(async (req, res) => {
    const userId = req.user!.id;
    const plan = await planService.getCurrent(userId);
    res.json({ plan });
  }),
};
```

**Service** business logic:
```ts
// services/plan.service.ts
export const planService = {
  async generateForUser(userId: string): Promise<Plan> {
    const profile = await profileRepo.findByUserId(userId);
    if (!profile) throw new ApiError(404, 'PROFILE_NOT_FOUND', 'Profil belum dibuat');

    // Panggil ML untuk workout & meal
    const [workoutResult, mealResult] = await Promise.all([
      mlClient.workout.infer({ profile }),
      mlClient.meal.infer({ profile }),
    ]);

    // Persist plan + child rows
    const plan = await planRepo.create({
      userId,
      weekNumber: await planRepo.nextWeekNumber(userId),
      workoutDays: workoutResult.days,
      mealDays: mealResult.days,
    });

    return plan;
  },

  async getCurrent(userId: string): Promise<Plan> {
    const plan = await planRepo.findCurrent(userId);
    if (!plan) throw new ApiError(404, 'NO_ACTIVE_PLAN', 'Belum ada rencana aktif');
    return plan;
  },
};
```

**Repository** DB access:
```ts
// repositories/plan.repo.ts
import { prisma } from '../config/db';

export const planRepo = {
  async findCurrent(userId: string) {
    return prisma.weeklyPlan.findFirst({
      where: { userId, archivedAt: null },
      include: { workoutDays: { include: { exercises: true } }, mealDays: { include: { meals: { include: { foods: true } } } } },
      orderBy: { startDate: 'desc' },
    });
  },

  async create(input: CreatePlanInput) {/* ... */},

  async nextWeekNumber(userId: string) {
    const last = await prisma.weeklyPlan.findFirst({ where: { userId }, orderBy: { weekNumber: 'desc' } });
    return (last?.weekNumber ?? 0) + 1;
  },
};
```

## 4. Aturan Import

- Controllers boleh import: services, validators, middleware, utils.
- Services boleh import: repositories, ml-client, services lain, utils.
- **Services TIDAK boleh import dari controllers atau Express types.**
- Repositories hanya import: Prisma client, types.
- Middleware boleh import: utils, services (jarang), Express types.

## 5. Setup Express App

File: `src/app.ts`

```ts
import express from 'express';
import cors from 'cors';
import helmet from 'helmet';
import { errorMiddleware } from './middleware/error.middleware';
import { requestLogger } from './middleware/request-logger.middleware';
import { v1Router } from './routes';
import { env } from './config/env';
import { corsOptions } from './config/cors';

export function createApp() {
  const app = express();

  app.use(helmet());
  app.use(cors(corsOptions));
  app.use(express.json({ limit: '1mb' }));
  app.use(express.urlencoded({ extended: true }));
  app.use(requestLogger);

  app.get('/health', (_, res) => res.json({ status: 'ok', timestamp: new Date().toISOString() }));

  app.use('/v1', v1Router);

  app.use(errorMiddleware);   // harus paling akhir
  return app;
}
```

File: `src/server.ts`

```ts
import { createApp } from './app';
import { env } from './config/env';
import { logger } from './utils/logger';
import { startCronJobs } from './jobs/cron';

const app = createApp();

const server = app.listen(env.PORT, () => {
  logger.info({ port: env.PORT }, `Heltigo API listening on :${env.PORT}`);
});

startCronJobs();

// Graceful shutdown
process.on('SIGTERM', () => {
  logger.info('SIGTERM received, shutting down...');
  server.close(() => process.exit(0));
});
```

## 6. Async Handler Wrapper

Hindari try/catch berulang di controllers:

```ts
// middleware/async.middleware.ts
import { Request, Response, NextFunction, RequestHandler } from 'express';

export const asyncHandler = (fn: RequestHandler): RequestHandler =>
  (req, res, next) => Promise.resolve(fn(req, res, next)).catch(next);
```

Dengan ini, error otomatis lewat ke `errorMiddleware`.

## 7. ApiError Custom Class

```ts
// utils/api-error.ts
export class ApiError extends Error {
  constructor(
    public statusCode: number,
    public code: string,
    message: string,
    public details?: Record<string, unknown>
  ) {
    super(message);
    this.name = 'ApiError';
  }
}
```

`errorMiddleware`:
```ts
export function errorMiddleware(err: Error, req: Request, res: Response, _next: NextFunction) {
  if (err instanceof ApiError) {
    return res.status(err.statusCode).json({
      error: { code: err.code, message: err.message, details: err.details },
    });
  }
  // Zod error
  if (err instanceof ZodError) {
    return res.status(400).json({
      error: { code: 'VALIDATION_ERROR', message: 'Input tidak valid', details: err.format() },
    });
  }
  logger.error({ err }, 'Unhandled error');
  res.status(500).json({ error: { code: 'INTERNAL_ERROR', message: 'Terjadi kesalahan internal' } });
}
```
